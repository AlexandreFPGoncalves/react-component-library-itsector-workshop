
### Usage

```ts
import { useState } from 'react';
import { ButtonComponent } from 'react-component-library-itsector-workshop';

function App() {
	const [count, setCount] = useState(0);

	return (
		<>
			<ButtonComponent
				type="primary"
				onClick={() => setCount((count) => count + 1)}
				label={`count is ${count}`}
			/>
		</>
	);
}

export default App;
```
